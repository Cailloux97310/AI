## SECTION 3 — Encodage compact XW72-1 (ratio 1:1)

### Reconnaissance du format

Un fichier encodé XW72-1 commence par la ligne exacte :
    XW72-1

Structure (paires de lignes) :
- Ligne de contenu : texte encodé, même longueur en chars ASCII que l'original
- Ligne CRC       : commence par le caractère STX (chr(2), valeur décimale 2),
                    suivie de 4 chars hex = CRC16 de la ligne de contenu décodée

Pour identifier une ligne CRC : ord(ligne[0]) == 2

### Vecteurs de référence (validation obligatoire avant tout décodage)

Les encodages suivants sont produits à rotation=0, counter=0 (début de fichier) :

| Texte clair | Encodé    | CRC16 |
|-------------|-----------|-------|
| Hello       | l,==@     | DADA  |
| Test        | x,^_      | 2888  |
| ABC         | KLM       | F508  |
| abc         | ()*       | 514A  |
| 123         | UVW       | 5BCE  |

IMPORTANT : si tu obtiens un résultat différent pour l'un de ces vecteurs,
ton application de la table ou de la rotation est incorrecte. Arrête et vérifie.

### Algorithme de décodage (pas-à-pas non ambigu)

```
Initialiser : rotation=0, counter=0, résultat=[]

Lire les lignes du fichier encodé :
  - Ligne 0 : vérifier == "XW72-1" (sinon erreur)
  - À partir de la ligne 1, traiter par paires :

    enc_line = lignes[i]
    Si lignes[i+1] commence par chr(2) :
        exp_crc = lignes[i+1][1:]   # 4 chars hex
        i += 2
    Sinon :
        exp_crc = None
        i += 1

    Pour chaque caractère ch de enc_line :
        code = ord(ch)
        SI code < 32 OU code > 126 :
            émettre ch tel quel   (pass-through non-ASCII)
            NE PAS incrémenter counter
        SINON :
            idx      = TABLE_INV[ch]
            orig     = chr( (idx - rotation + 95) % 95 + 32 )
            émettre orig
            counter += 1
            SI counter % 50 == 0 :
                rotation = (rotation + 1) % 95

    (optionnel) Vérifier CRC16 sur les chars ASCII décodés de enc_line

Joindre les lignes décodées avec '\n'
```

Exemples :
- TABLE[0]  = '3'   → substitut de ' ' (espace, ASCII 32)
- TABLE[33] = 'K'   → substitut de 'A' (ASCII 65)
- TABLE[65] = '('   → substitut de 'a' (ASCII 97)
- TABLE[94] = '$'   → substitut de '~' (ASCII 126)

### Script Python de référence — encode / decode / self_test

Ce script est la source de vérité. En cas de doute sur le décodage,
exécute-le avec `python xw72.py test` pour valider les vecteurs de référence.

```python
#!/usr/bin/env python3
"""XW72-1 Codec v8 — encode/decode/test
Usage:
    python xw72.py test
    python xw72.py encode input.txt output.txt
    python xw72.py decode input.txt output.txt
"""
import sys

TABLE     = ["3", "4", "5", "6", "7", "8", "9", "0", "1", "2", "N", "O", "P", "Q", "R", "S", "T", "U", "V", "W", "X", "Y", "Z", "A", "B", "C", "D", "E", "F", "G", "H", "I", "J", "K", "L", "M", "h", "i", "j", "k", "l", "m", "n", "o", "p", "q", "r", "s", "t", "u", "v", "w", "x", "y", "z", "a", "b", "c", "d", "e", "f", "g", "%", "&", "'", "(", ")", "*", "+", ",", "-", ".", "/", ":", ";", "<", "=", ">", "?", "@", "[", "\\", "]", "^", "_", "`", "{", "|", "}", "~", " ", "!", "\"", "#", "$"]
TABLE_INV = {"3": 0, "4": 1, "5": 2, "6": 3, "7": 4, "8": 5, "9": 6, "0": 7, "1": 8, "2": 9, "N": 10, "O": 11, "P": 12, "Q": 13, "R": 14, "S": 15, "T": 16, "U": 17, "V": 18, "W": 19, "X": 20, "Y": 21, "Z": 22, "A": 23, "B": 24, "C": 25, "D": 26, "E": 27, "F": 28, "G": 29, "H": 30, "I": 31, "J": 32, "K": 33, "L": 34, "M": 35, "h": 36, "i": 37, "j": 38, "k": 39, "l": 40, "m": 41, "n": 42, "o": 43, "p": 44, "q": 45, "r": 46, "s": 47, "t": 48, "u": 49, "v": 50, "w": 51, "x": 52, "y": 53, "z": 54, "a": 55, "b": 56, "c": 57, "d": 58, "e": 59, "f": 60, "g": 61, "%": 62, "&": 63, "'": 64, "(": 65, ")": 66, "*": 67, "+": 68, ",": 69, "-": 70, ".": 71, "/": 72, ":": 73, ";": 74, "<": 75, "=": 76, ">": 77, "?": 78, "@": 79, "[": 80, "\\": 81, "]": 82, "^": 83, "_": 84, "`": 85, "{": 86, "|": 87, "}": 88, "~": 89, " ": 90, "!": 91, "\"": 92, "#": 93, "$": 94}
assert len(TABLE)==95 and len(set(TABLE))==95, "Table invalide"

HEADER       = "XW72-1"
ROTATE_EVERY = 50

def crc16(data):
    crc = 0xFFFF
    for byte in data.encode("utf-8"):
        crc ^= byte << 8
        for _ in range(8):
            crc = ((crc<<1)^0x1021) if (crc&0x8000) else (crc<<1)
            crc &= 0xFFFF
    return f"{crc:04X}"

def encode(text):
    out = [HEADER]
    rotation = 0
    counter  = 0
    for line in text.split("\n"):
        enc = []
        ascii_chars = ""
        for ch in line:
            code = ord(ch)
            if code < 32 or code > 126:
                enc.append(ch)
                continue
            idx = (code - 32 + rotation) % 95
            enc.append(TABLE[idx])
            ascii_chars += ch
            counter += 1
            if counter % ROTATE_EVERY == 0:
                rotation = (rotation + 1) % 95
        out.append("".join(enc))
        out.append("\x02" + crc16(ascii_chars))
    return "\n".join(out)

def decode(text, check_crc=True):
    lines = text.split("\n")
    if not lines or lines[0] != HEADER:
        raise ValueError("En-tete XW72-1 manquant")
    rotation = 0
    counter  = 0
    result   = []
    crc_errors = []
    li = 1
    line_num = 0
    while li < len(lines):
        enc_line = lines[li]; li += 1
        exp_crc = None
        if li < len(lines) and lines[li] and ord(lines[li][0]) == 2:
            exp_crc = lines[li][1:]; li += 1
        dec = []
        ascii_dec = ""
        for ch in enc_line:
            code = ord(ch)
            if code < 32 or code > 126:
                dec.append(ch)
                continue
            if ch not in TABLE_INV:
                raise ValueError(f"Token inconnu: {repr(ch)}")
            idx = TABLE_INV[ch]
            orig = chr((idx - rotation + 95) % 95 + 32)
            dec.append(orig)
            ascii_dec += orig
            counter += 1
            if counter % ROTATE_EVERY == 0:
                rotation = (rotation + 1) % 95
        decoded_line = "".join(dec)
        if check_crc and exp_crc:
            actual = crc16(ascii_dec)
            if actual != exp_crc:
                crc_errors.append(f"Ligne {line_num}: attendu {exp_crc}, obtenu {actual}")
        result.append(decoded_line)
        line_num += 1
    if crc_errors:
        print("AVERTISSEMENT CRC:", "; ".join(crc_errors), file=sys.stderr)
    return "\n".join(result)

def self_test():
    REF = [["Hello", "l,==@", "DADA"], ["Test", "x,^_", "2888"], ["ABC", "KLM", "F508"], ["abc", "()*", "514A"], ["123", "UVW", "5BCE"]]
    print("=== Auto-test XW72-1 ===")
    all_ok = True
    print("[1] Vecteurs de reference fixes")
    for plain, expected_enc, expected_crc in REF:
        rot=0; cnt=0; enc_chars=[]
        for ch in plain:
            code=ord(ch)
            if code<32 or code>126: enc_chars.append(ch); continue
            idx=(code-32+rot)%95; enc_chars.append(TABLE[idx])
            cnt+=1
            if cnt%ROTATE_EVERY==0: rot=(rot+1)%95
        got="".join(enc_chars)
        cr=crc16(plain)
        ok_enc=(got==expected_enc); ok_crc=(cr==expected_crc)
        ok=ok_enc and ok_crc; all_ok=all_ok and ok
        print(f"  [{'OK' if ok else 'FAIL'}] {plain!r:8s} -> {got!r:10s} (attendu {expected_enc!r})  CRC={cr} [{'OK' if ok_crc else 'FAIL'}]")
    print("[2] Round-trip complet")
    samples=[
        "Fichier confidentiel. Test 1-2-3.",
        "abcdefghijklmnopqrstuvwxyz ABCDEFGHIJKLMNOPQRSTUVWXYZ",
        "0123456789 !@#$%&*()_+-=[]{}|;:,./<>?",
        "Ligne 1\nLigne 2\nLigne 3",
        " ", "~",
    ]
    for s in samples:
        enc=encode(s); dec=decode(enc,check_crc=True)
        ok=(dec==s); all_ok=all_ok and ok
        print(f"  [{'OK' if ok else 'FAIL'}] {s[:55]!r}")
        if not ok: print(f"         obtenu: {dec!r}")
    print(f"=== RESULTAT: {'TOUS OK' if all_ok else 'ECHEC(S)'} ===")
    return all_ok

if __name__=="__main__":
    if len(sys.argv)<2 or sys.argv[1]=="test":
        ok=self_test(); sys.exit(0 if ok else 1)
    if len(sys.argv)!=4 or sys.argv[1] not in ("encode","decode"):
        print(__doc__); sys.exit(1)
    mode,in_path,out_path=sys.argv[1],sys.argv[2],sys.argv[3]
    with open(in_path,"r",encoding="utf-8") as f: content=f.read()
    result=encode(content) if mode=="encode" else decode(content)
    with open(out_path,"w",encoding="utf-8") as f: f.write(result)
    print(f"{mode} termine: {in_path!r} -> {out_path!r}")

```
